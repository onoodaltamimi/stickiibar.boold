<script>
let steps = [];
let step = 0;

function play() {
    document.getElementById("clickSound").play();
}

function startFabric() {
    steps = [
        ["اختر الاستكرات", "./1.png"],
        ["ضع الباوتش على الحصيرة", "./2.png"],
        ["رتب الاستكرات", "./3.png"],
        ["ضع ورق الحماية", "./4.png"],
        ["اضغط بالمكبس الحراري", "./5.png"],
        ["انتظر حتى يبرد", "./6.png"],
        ["أزل الطبقة الشفافة", "./7.png"],
        ["انتهيت! استمتع بقطعتك 🎉", "./finish.png"]
    ];
    startSteps();
}

function startCup() {
    steps = [
        ["اختر الاستكر", "./1.png"],
        ["ضع الاستكر واضغط عليه جيداً", "./2.png"],
        ["أزل الطبقة الشفافة", "./3.png"],
        ["انتهيت! استمتع بقطعتك 🎉", "./finish.png"]
    ];
    startSteps();
}

function startSteps() {
    step = 0;
    document.getElementById("start").style.display = "none";
    document.getElementById("stepArea").classList.remove("hidden");
    show();
}

function show() {
    document.getElementById("stepText").innerText = steps[step][0];
    document.getElementById("stepImage").src = steps[step][1];
    if(step > 0){
        document.getElementById("backBtn").classList.remove("hidden");
    } else {
        document.getElementById("backBtn").classList.add("hidden");
    }
    if(step === steps.length - 1){
        document.querySelector(".next").classList.add("hidden");
    } else {
        document.querySelector(".next").classList.remove("hidden");
    }
}

function next() {
    play();
    if(step < steps.length - 1){
        step++;
        show();
    }
}

function prev() {
    play();
    if(step > 0){
        step--;
        show();
    }
}
</script>
